  subroutine destroy_key(self)
  !---------------------------------------------------------------------------------------------------------------------------------
  !< Destroy the node key.
  !---------------------------------------------------------------------------------------------------------------------------------
  class(dictionary_node), intent(inout) :: self !< The node.
  !---------------------------------------------------------------------------------------------------------------------------------

  !---------------------------------------------------------------------------------------------------------------------------------
  if (allocated(self%key)) then
    associate(key=>self%key)
      select type(key)
      class is(key_base)
        call key%destroy
      endselect
    endassociate
    deallocate(self%key)
  endif
  !---------------------------------------------------------------------------------------------------------------------------------
  endsubroutine destroy_key
